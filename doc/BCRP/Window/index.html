class CSVHandler():
    """
    Contains common CSV usage files to use across other classes.
    """

    def read_csv(self, path):
        """
        Reads a CSV specified with 'path' and returns an array
        containing the data.
        """


        data = []  # Array to store the column data


        with open(path, 'r') as file:
            for column_index in range(1, 5):
                field = []
                file.seek(0)  # Reset the file pointer to the beginning
                next(file)

                for line in file:
                    row = line.strip().split(',')
                    if len(row) > column_index:
                        val = float(row[column_index])
                        if val > 80:
                            val = 80
                        if val < -80:
                            val = -80

                        field.append(val)
                data.append(field)
        
        return data

class FrameBase(Frame):

    """
    A class which all frames extend.
    Contains an initializer and styling definitions
    to maintain a constant style throughout the app.
    """


    # Constructor
    def __init__(self, master: Tk, parent_window):

        # Initialise global styles for tkinter objects
        self.init_styles()

        # Calls the Tk Frame Constructor
        super().__init__(master)

        self.config(background="white")

        # All widgets should be stored in the self.widgets dictionary
        self.widgets = dict()

        # All frames should be stored in the self.frames dictionary
        self.frames = dict()

        # The window which holds the frames (MHostWindow)
        self.parent_window = parent_window
        


    def init_styles(self):
        """
        Initialise the styles used throughout the program.
        """

        # Define button styles
        self.button_style = {
            "font": ("Bahnschrift Light", 12),
            "bg": "#EEEEEE",
            "fg": "black",
            "width": 22,
            "height": 2,
            "bd": 1,
            "highlightthickness": 0,
            "relief": "ridge",
        }

        self.small_button_style = {
            "font": ("Bahnschrift Light", 10),
            "bg": "#EEEEEE",
            "fg": "black",
            "width": 15,
            "height": 1,
            "bd": 1,
            "highlightthickness": 0,
            "relief": "ridge",
        }

        # Define title styles
        self.title_label_style = {
            "font":("Bahnschrift", 26, "bold"),
            "fg":"#3C79BA",
            "bg":"white",
            
        }

        # Define label styles
        self.secondary_label_style = {
            "font":("Bahnschrift Light", 14), 
            "fg":"#424242",
            "bg":"white",
        }

        self.tertiary_label_style = {
            "font": ("Bahnschrift Light", 9),
            "fg":"#343434",
            "bg":"white",
            
        }

        self.image_style = {
            "bg":"white",
            "bd":0
        }


    def forget_frame(self):
        """
        Forget all subframes in a frame, then forget the frame itself.
        """
        for x in self.frames.values():
            x.forget()
        self.forget()

class FrameCSVDisplay(FrameBase):
    """
    A frame which displays the current CSV to the screen.
    """
    def __init__(self, master:Tk, main_window, frame, simulation_frame, path):
        
        
        self.path = path
        self.frame = frame
        
        self.simulation_frame = simulation_frame
        self.simulation_frame.penSpawn()

        self.play_animation = None
        self.pause_animation = True

        super().__init__(master, main_window)

        self.pack_propagate(0)

        csv_handler = CSVHandler()

        self.data = csv_handler.read_csv(self.path)

        self.create_text()

        self.create_canvas()

        self.create_scrollbar()

        self.create_buttons()

        self.plot_graph()



    def play_csv(self):
        """
        Plays the current CSV, incrementing the slider and starting the associated simulation.
        """
        

        self.plus_csv(1)

        if self.pause_animation:
            self.pause_animation = False
            self.animate_csv()

        self.simulation_frame.start_simulation(self.data)


    def pause_csv(self):
        """
        Pauses the CSV, pausing the slider and the associated simulation.
        """
        self.pause_animation = True
        self.simulation_frame.penStop()

    def animate_csv(self):
        """
        Visually increments the slider on the screen.
        """
        # Update the position of the vertical line
        x = self.canvas.coords(self.vertical_line)[0] + 1
        self.canvas.coords(self.vertical_line, x, -50, x, 300)

        # Repeat the animation if not paused
        if not self.pause_animation:
            self.after(10, self.animate_csv)

    def create_text(self):
        """
        Draws labels depicting the fields of the CSV on the left side of the screen.
        """
        # Create the labels frame
        labels_frame = Frame(self)
        labels_frame.pack(side="left", fill="y")

        # Create the labels
        label_1 = Label(labels_frame, text="TP9")
        label_1.pack(anchor="w", pady=35, padx=(10,0))

        label_2 = Label(labels_frame, text="AF7")
        label_2.pack(anchor="w", pady=35, padx=(10,0))

        label_3 = Label(labels_frame, text="AF8")
        label_3.pack(anchor="w", pady=35, padx=(10,0))

        label_4 = Label(labels_frame, text="TP10")
        label_4.pack(anchor="w", pady=35, padx=(10,0))
        pass

    def create_canvas(self):
        """
        Creates the canvas widget which will hold the CSV.
        """
        # Create a canvas widget and add it to the frame
        self.canvas = Canvas(self)
        self.canvas.pack(fill="both", expand=True)

        # Draw the initial vertical line
        x = 50  # x-coordinate of the vertical line
        self.vertical_line = self.canvas.create_line(x, -50, x, 300, fill="black")

    def create_scrollbar(self):
        """
        Creates the scrollbar used to view different parts of the CSV.
        """
        # Create a scrollbar widget and add it to the frame
        self.scrollbar = Scrollbar(self, orient="horizontal", command=self.canvas.xview)
        self.scrollbar.pack(fill="x")


    
    def create_buttons(self):
        """
        Creates the buttons used to change the position of the CSV.
        """

        # Create buttons
        self.play_image = PhotoImage(file="./Assets/play.png")
        self.pause_image = PhotoImage(file="./Assets/pause.png")
        self.rewind_image = PhotoImage(file="./Assets/rewind.png")
        self.minus_image = PhotoImage(file="./Assets/minus.png")
        self.plus_image = PhotoImage(file="./Assets/plus.png")


        self.back_button = Button(self, text="Back", **self.button_style, command=lambda:self.frame.change_csv_frame())
        self.play_button = Button(self, image=self.play_image, command = lambda:self.play_csv())
        self.pause_button = Button(self, image=self.pause_image, command = lambda:self.pause_csv())
        self.rewind_button = Button(self, image=self.rewind_image, command = lambda:self.rewind_csv())
        self.minus_button = Button(self, image=self.minus_image, command = lambda:self.plus_csv(-10))
        self.plus_button = Button(self, image=self.plus_image, command = lambda:self.plus_csv(10))

        # Pack buttons to the right
        self.play_button.pack(side="left", fill="both")
        self.pause_button.pack(side="left", fill="both")
        self.rewind_button.pack(side="left", fill="both")
        self.back_button.pack(side="left", fill="both", expand=True)
        self.minus_button.pack(side="left", fill="both")
        self.plus_button.pack(side="left", fill="both")

        """

        self.canvas.coords(self.vertical_line, x, -50, x, 300)

        for column_index in range(0, num_columns):
            
            #self.canvas.create_text(text_x, text_y, text=text[column_index], anchor="e")

            self.canvas.configure(scrollregion=self.canvas.bbox("all"))

class FrameCSVList(FrameBase):
    """
    A frame which lists the CSVs in the ./CSVs directory onto the screen.
    """

    def __init__(self, master:Tk, main_window, parent_windows):


        self.parent_windows = parent_windows
        self.main_window = main_window
        
        super().__init__(master, main_window)

        self.pack_propagate(0)  # Disable automatic resizing
        
        self.configure(highlightthickness=1, highlightbackground="black")
        

        self.display_files()

    def get_file(self, path, name):
        """
        Retrieves the file information for a specified file.
        Retrieves the file name, file extension, modification time, and length.
        """
        
        file_stat = os.stat(path)
        new_file_name = name.split(".")[0]
        modification_time = datetime.datetime.fromtimestamp(file_stat.st_mtime).strftime("%d/%m/%y %H:%M:%S")
        file_extension = os.path.splitext(name)[1]
        file_length = file_stat.st_size
        file_info = {
            "name": new_file_name,
            "modification_time": modification_time,
            "extension": file_extension,
            "length": file_length
        }
        return file_info

    def get_file_info(self, file_name):
        """
        Retrieves the file information for a file.
        """

        file_path = os.path.join(self.directory_path, file_name)
        
        file_info = self.get_file(file_path, file_name)

        self.file_info_list.append(file_info)


            

    def display_files(self):

        self.delete_image = PhotoImage(file='./Assets/delete.png')

        self.directory_path = "./CSVs"

        self.file_info_list = []

        for file_name in os.listdir(self.directory_path):
            self.get_file_info(file_name)

        # set number of rows to number of csv files
        row_length = len(self.file_info_list)

        # set grid propagate to 0 to stop the frame from resizing
        self.grid_propagate(0)

        # set the 1st column to weight 1 to allow the canvas to fill the space
        self.columnconfigure(1,weight=1)
        self.rowconfigure(1, weight=1)

        self.canvas = Canvas(self)
        self.canvas.grid(row=0, column=0, sticky=NSEW, columnspan=2, rowspan=row_length+1)

        # Create a frame inside the canvas to hold the buttons
        self.frame = Frame(self.canvas)
        self.frame.grid(row=0, column=0, sticky=NSEW)

        # Create a scrollbar and associate it with the canvas
        scrollbar = Scrollbar(self, orient=VERTICAL, command=self.canvas.yview)

def get_file(self, path, name):
    """
    Retrieves the file information for a specified file.
    Retrieves the file name, file extension, modification time, and length.
    """
    
    file_stat = os.stat(path)
    new_file_name = name.split(".")[0]
    modification_time = datetime.datetime.fromtimestamp(file_stat.st_mtime).strftime("%d/%m/%y %H:%M:%S")
    file_extension = os.path.splitext(name)[1]
    file_length = file_stat.st_size
    file_info = {
        "name": new_file_name,
        "modification_time": modification_time,
        "extension": file_extension,
        "length": file_length
    }
    return file_info

class FrameSimulation(FrameBase):
    """
    A frame to allow the user to apply a simulation of the pen's trajectory using a specified CSV.
    """

    def __init__(self, master:Tk, main_window):
        super().__init__(master, main_window)
        self.pack_propagate(0)

        self.configure(highlightthickness=1, highlightbackground="black")

        self.color_index = 0 # set initial colour to red
        self.colors = [ # will change upon blinking
            "red",
            "orange",
            "yellow",
            "green",
            "light blue",
            "blue",
            "purple"
        ]
        
        self.DELTA_MIN = 1
        self.DELTA_MAX = 4 # range = 4
        self.THETA_MIN = 4
        self.THETA_MAX = 8 # range = 4
        self.ALPHA_MIN = 8
        self.ALPHA_MAX = 12 # range = 4
        self.BETA_MIN = 12
        self.BETA_MAX = 30 # range = 18

        self.BETA_RANGE = self.BETA_MAX - self.BETA_MIN

        self.forward = 3 # default speed of line
        self.stop = False
        
        self.loops = 0 # looping over the csv file
        
        
        self.canvas = Canvas(self)

        self.canvas.pack(fill=BOTH, expand=TRUE)
        self.pen = turtle.RawTurtle(self.canvas) # create pen in canvas 
        self.penSpawn() # spawns in top left corner
        self.spawnGUI()


    def start_simulation(self, csv):
        """
        Starts the simulation on press of the start button.
        """

        # starts when start button is pressed on display

        self.penLoop(csv)
    

    def penStop(self):
        """
        Stops the simulation on press of the pause button.
        """
        self.stop = True # when stop button is pressed


    def penSpawn(self):
        """
        Spawns the pen to the screen in a specified position.
        """
        self.loops=0
        self.pen.clear()
        self.pen.speed(0)
        self.pen.width(2)
        self.pen.color('red')
        self.pen.penup()
        self.pen.goto(10,0) # initially the pen will start in top left
        self.pen.pendown()
        self.pen.setheading(0)


    def oscilate(self): # causes the pen to oscillate based on the delta level
        """
        Controls the current oscillation of the pen.
        """
        # delta : 1 - 4 : oscillation

        percentage = (self.avgValue - self.DELTA_MIN) * (100 / (self.DELTA_MAX - self.avgValue))
        # convert the beta value to a percentage / 100

        self.max = percentage / 3 # highest the curve will go
        self.low = -percentage / 3 # lowest the curve will go

        if self.current >= self.max:
            self.dir = "DOWN" # to reach the minimum point
        elif self.current <= self.low:
            self.dir = "UP" # to reach the maximum point

        if self.dir == "UP":
            self.current = self.current + 1
            self.pen.left(3)

        elif self.dir == "DOWN":
            self.current = self.current - 1
            self.pen.right(3)

    def thickness(self): # causes the pen thickness to change based on beta level
        """
        Controls the current thickness of the pen.
        """
        # beta : 12 - 30 : thickness

        percentage = (self.avgValue - self.BETA_MIN) * (100 / (self.BETA_MAX - self.avgValue))
        # convert the beta value to a percentage / 100

        if self.avgValue > self.BETA_MAX + (self.BETA_RANGE / 2): # pen thickess increases
            self.pen.width(((percentage - 60)//5) + 1)
            self.dot_timer = 0
            self.pen.pendown()

        elif self.avgValue < (self.BETA_MAX + self.BETA_MIN) / 2: # line becomes dotted
            self.pen.width(2)
            self.dot_timer = ((45 - percentage)//10)
        else:
            self.pen.width(2) # line is normal
            self.dot_timer = 0
            self.pen.pendown()


    def penBlink(self): # pen changes colour on blink
        """
        Controls the current colour of the pen.
        """
        self.color_index =  self.color_index + 1 # goes to next colour in rainbow
        self.pen.color(self.colors[self.color_index % len(self.colors)])

    def penLoop(self, csv):
        """
        The operations performed by the pen on every loop.
        """
        # default values
        self.stop = False
        self.dot_timer = 0
        self.dot_current = 0
        self.max, self.low = 0, 0
        self.current = 0
        self.dir = "DOWN"
        self.spike=False

        while(self.stop != True): # loops through values to see if they have changed
            
            # beta : 12 - 30 : thickness
            # alpha : 8 - 12 : left
            # theta : 4 - 8 : right
            # delta : 1 - 4 : oscillation

            
            self.loops += 1
            self.avgValue = ((csv[0][self.loops] + csv[1][self.loops] + csv[2][self.loops] + csv[3][self.loops]) / 4)


            if(self.avgValue <= self.BETA_MAX and self.avgValue > self.BETA_MIN):
                self.thickness()
                self.spike=False
            
            if(self.avgValue <= self.DELTA_MAX and self.avgValue > self.DELTA_MIN):
                self.oscilate()
                self.spike=False

            if(self.avgValue <= self.THETA_MAX and self.avgValue > self.THETA_MIN):
                self.pen.right((self.avgValue - self.THETA_MIN) * (90 / (self.THETA_MAX - self.avgValue))) 
                self.spike=False
                # converts the value into a degree from 0 to 90
            
            if(self.avgValue <= self.ALPHA_MAX and self.avgValue > self.ALPHA_MIN):
                self.pen.left((self.avgValue - self.ALPHA_MIN) * (90 / (self.ALPHA_MAX - self.avgValue))) 
                self.spike=False
                # converts the value into a degree from 0 to 90

            if((self.avgValue > 70 or self.avgValue < -70) and self.spike==False):
                self.spike = True
                self.penBlink()
     
            # Handling dotted line (meditation wave)
            if self.dot_timer > 0:
                self.pen.penup()
                self.dot_current = self.dot_current + 0.25
                if self.dot_current >= self.dot_timer:
                    self.pen.pendown()
                    self.pen.forward(self.forward)
                    self.dot_current = 0
                    continue
                else:
                    pass
                
            self.pen.forward(self.forward)

            
    def spawnGUI(self):
        """
        Creates the interface for the pen to draw onto.
        """
        self.interface = dict()
        button_font = ("Arial 16 bold")

    Controls the current oscillation of the pen.
    """
    # delta : 1 - 4 : oscillation

    percentage = (self.avgValue - self.DELTA_MIN) * (100 / (self.DELTA_MAX - self.avgValue))
    # convert the beta value to a percentage / 100

    self.max = percentage / 3 # highest the curve will go
    self.low = -percentage / 3 # lowest the curve will go

    if self.current >= self.max:
        self.dir = "DOWN" # to reach the minimum point
    elif self.current <= self.low:
        self.dir = "UP" # to reach the maximum point

    if self.dir == "UP":
        self.current = self.current + 1
        self.pen.left(3)

    elif self.dir == "DOWN":
        self.current = self.current - 1
        self.pen.right(3)

def penLoop(self, csv):
    """
    The operations performed by the pen on every loop.
    """
    # default values
    self.stop = False
    self.dot_timer = 0
    self.dot_current = 0
    self.max, self.low = 0, 0
    self.current = 0
    self.dir = "DOWN"
    self.spike=False

    while(self.stop != True): # loops through values to see if they have changed
        
        # beta : 12 - 30 : thickness
        # alpha : 8 - 12 : left
        # theta : 4 - 8 : right
        # delta : 1 - 4 : oscillation

        
        self.loops += 1
        self.avgValue = ((csv[0][self.loops] + csv[1][self.loops] + csv[2][self.loops] + csv[3][self.loops]) / 4)


        if(self.avgValue <= self.BETA_MAX and self.avgValue > self.BETA_MIN):
            self.thickness()
            self.spike=False
        
        if(self.avgValue <= self.DELTA_MAX and self.avgValue > self.DELTA_MIN):
            self.oscilate()
            self.spike=False

        if(self.avgValue <= self.THETA_MAX and self.avgValue > self.THETA_MIN):
            self.pen.right((self.avgValue - self.THETA_MIN) * (90 / (self.THETA_MAX - self.avgValue))) 
            self.spike=False
            # converts the value into a degree from 0 to 90
        
        if(self.avgValue <= self.ALPHA_MAX and self.avgValue > self.ALPHA_MIN):
            self.pen.left((self.avgValue - self.ALPHA_MIN) * (90 / (self.ALPHA_MAX - self.avgValue))) 
            self.spike=False
            # converts the value into a degree from 0 to 90

        if((self.avgValue > 70 or self.avgValue < -70) and self.spike==False):
            self.spike = True
            self.penBlink()
 
        # Handling dotted line (meditation wave)
        if self.dot_timer > 0:
            self.pen.penup()
            self.dot_current = self.dot_current + 0.25
            if self.dot_current >= self.dot_timer:
                self.pen.pendown()
                self.pen.forward(self.forward)
                self.dot_current = 0
                continue
            else:
                pass
            
        self.pen.forward(self.forward)

def thickness(self): # causes the pen thickness to change based on beta level
    """
    Controls the current thickness of the pen.
    """
    # beta : 12 - 30 : thickness

    percentage = (self.avgValue - self.BETA_MIN) * (100 / (self.BETA_MAX - self.avgValue))
    # convert the beta value to a percentage / 100

    if self.avgValue > self.BETA_MAX + (self.BETA_RANGE / 2): # pen thickess increases
        self.pen.width(((percentage - 60)//5) + 1)
        self.dot_timer = 0
        self.pen.pendown()

    elif self.avgValue < (self.BETA_MAX + self.BETA_MIN) / 2: # line becomes dotted
        self.pen.width(2)
        self.dot_timer = ((45 - percentage)//10)
    else:
        self.pen.width(2) # line is normal
        self.dot_timer = 0
        self.pen.pendown()

class HostWindow:
    """
    The window which all frames are displayed on.
    """

    # Constructor
    def __init__(self):

        self.init_frame_dictionary()

        # Initiating the Tk Instance
        self.master = Tk()
        self.master.attributes("-fullscreen", True)
        self.master.configure(background="white")  
        self.master.title("Brain Controlled Robot Painting")
        self.frame = FrameMainMenu(self.master, self)
        self.frame.pack()


    def init_frame_dictionary(self):
        """
        Initialise a dictionary converting classes to Strings to avoid circular imports.
        """
        self.names_to_frames = {
        "main menu":FrameMainMenu,
        "virtual menu": FrameVirtualMenu,
        "physical menu": FramePhysicalMenu,
        "settings": FrameSettings,
        "account menu": FrameAccountMenu,
        "csv list": FrameCSVList,
        "csv display": FrameCSVDisplay
        }

    def change_frame(self, new_frame_name):
        """
        Switches the currently hosted frame.
        """
        self.frame.forget_frame()
        frame = self.names_to_frames.get(new_frame_name)
        self.frame = frame(self.master, self)
        self.frame.pack()
