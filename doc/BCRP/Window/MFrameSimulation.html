import turtle
from Window import *
from tkinter import *

class FrameSimulation(FrameBase):
    """
    A frame to allow the user to apply a simulation of the pen's trajectory using a specified CSV.
    """

    def __init__(self, master:Tk, main_window):
        super().__init__(master, main_window)
        self.pack_propagate(0)

        self.configure(highlightthickness=1, highlightbackground="black")

        self.color_index = 0 # set initial colour to red
        self.colors = [ # will change upon blinking
            "red",
            "orange",
            "yellow",
            "green",
            "light blue",
            "blue",
            "purple"
        ]
        
        self.DELTA_MIN = 1
        self.DELTA_MAX = 4 # range = 4
        self.THETA_MIN = 4
        self.THETA_MAX = 8 # range = 4
        self.ALPHA_MIN = 8
        self.ALPHA_MAX = 12 # range = 4
        self.BETA_MIN = 12
        self.BETA_MAX = 30 # range = 18

        self.BETA_RANGE = self.BETA_MAX - self.BETA_MIN

        self.forward = 3 # default speed of line
        self.stop = False
        
        self.loops = 0 # looping over the csv file
        
        
        self.canvas = Canvas(self)

        self.canvas.pack(fill=BOTH, expand=TRUE)
        self.pen = turtle.RawTurtle(self.canvas) # create pen in canvas 
        self.penSpawn() # spawns in top left corner
        self.spawnGUI()


    def start_simulation(self, csv):
        """
        Starts the simulation on press of the start button.
        """

        # starts when start button is pressed on display

        self.penLoop(csv)
    

    def penStop(self):
        """
        Stops the simulation on press of the pause button.
        """
        self.stop = True # when stop button is pressed


    def penSpawn(self):
        """
        Spawns the pen to the screen in a specified position.
        """
        self.loops=0
        self.pen.clear()
        self.pen.speed(0)
        self.pen.width(2)
        self.pen.color('red')
        self.pen.penup()
        self.pen.goto(10,0) # initially the pen will start in top left
        self.pen.pendown()
        self.pen.setheading(0)


    def oscilate(self): # causes the pen to oscillate based on the delta level
        """
        Controls the current oscillation of the pen.
        """
        # delta : 1 - 4 : oscillation

        percentage = (self.avgValue - self.DELTA_MIN) * (100 / (self.DELTA_MAX - self.avgValue))
        # convert the beta value to a percentage / 100

        self.max = percentage / 3 # highest the curve will go
        self.low = -percentage / 3 # lowest the curve will go

        if self.current >= self.max:
            self.dir = "DOWN" # to reach the minimum point
        elif self.current <= self.low:
            self.dir = "UP" # to reach the maximum point

        if self.dir == "UP":
            self.current = self.current + 1
            self.pen.left(3)

        elif self.dir == "DOWN":
            self.current = self.current - 1
            self.pen.right(3)

    def thickness(self): # causes the pen thickness to change based on beta level
        """
        Controls the current thickness of the pen.
        """
        # beta : 12 - 30 : thickness

        percentage = (self.avgValue - self.BETA_MIN) * (100 / (self.BETA_MAX - self.avgValue))
        # convert the beta value to a percentage / 100

        if self.avgValue > self.BETA_MAX + (self.BETA_RANGE / 2): # pen thickess increases
            self.pen.width(((percentage - 60)//5) + 1)
            self.dot_timer = 0
            self.pen.pendown()

        elif self.avgValue < (self.BETA_MAX + self.BETA_MIN) / 2: # line becomes dotted
            self.pen.width(2)
            self.dot_timer = ((45 - percentage)//10)
        else:
            self.pen.width(2) # line is normal
            self.dot_timer = 0
            self.pen.pendown()


    def penBlink(self): # pen changes colour on blink
        """
        Controls the current colour of the pen.
        """
        self.color_index =  self.color_index + 1 # goes to next colour in rainbow
        self.pen.color(self.colors[self.color_index % len(self.colors)])

    def penLoop(self, csv):
        """
        The operations performed by the pen on every loop.
        """
        # default values
        self.stop = False
        self.dot_timer = 0
        self.dot_current = 0
        self.max, self.low = 0, 0
        self.current = 0
        self.dir = "DOWN"
        self.spike=False

        while(self.stop != True): # loops through values to see if they have changed
            
            # beta : 12 - 30 : thickness
            # alpha : 8 - 12 : left
            # theta : 4 - 8 : right
            # delta : 1 - 4 : oscillation

            
            self.loops += 1
            self.avgValue = ((csv[0][self.loops] + csv[1][self.loops] + csv[2][self.loops] + csv[3][self.loops]) / 4)


            if(self.avgValue <= self.BETA_MAX and self.avgValue > self.BETA_MIN):
                self.thickness()
                self.spike=False
            
            if(self.avgValue <= self.DELTA_MAX and self.avgValue > self.DELTA_MIN):
                self.oscilate()
                self.spike=False

            if(self.avgValue <= self.THETA_MAX and self.avgValue > self.THETA_MIN):
                self.pen.right((self.avgValue - self.THETA_MIN) * (90 / (self.THETA_MAX - self.avgValue))) 
                self.spike=False
                # converts the value into a degree from 0 to 90
            
            if(self.avgValue <= self.ALPHA_MAX and self.avgValue > self.ALPHA_MIN):
                self.pen.left((self.avgValue - self.ALPHA_MIN) * (90 / (self.ALPHA_MAX - self.avgValue))) 
                self.spike=False
                # converts the value into a degree from 0 to 90

            if((self.avgValue > 70 or self.avgValue < -70) and self.spike==False):
                self.spike = True
                self.penBlink()
     
            # Handling dotted line (meditation wave)
            if self.dot_timer > 0:
                self.pen.penup()
                self.dot_current = self.dot_current + 0.25
                if self.dot_current >= self.dot_timer:
                    self.pen.pendown()
                    self.pen.forward(self.forward)
                    self.dot_current = 0
                    continue
                else:
                    pass
                
            self.pen.forward(self.forward)

            
    def spawnGUI(self):
        """
        Creates the interface for the pen to draw onto.
        """
        self.interface = dict()
        button_font = ("Arial 16 bold")
